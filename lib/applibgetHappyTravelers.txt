import fs from "fs";
import path from "path";

export function getHappyTravelers() {
  const folder = path.join(
    process.cwd(),
    "public",
    "happy-travelers"
  );

  return fs
    .readdirSync(folder)
    .filter((file) =>
      /\.(jpg|jpeg|png|webp)$/i.test(file)
    )
    .map((file) => `/happy-travelers/${file}`);
}
